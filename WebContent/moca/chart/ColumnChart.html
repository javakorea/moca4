Column Chart

1.Gradient Color(Single column)

싱글 컬럼차트는 가장 심플한 컬럼차트의 종류이며 데이터 표현 방식이 직선적이며 단순하다.
이름에서 나타나듯이 싱글 컬럼 차트는 한개의 카테고리 안에서 다른 수치를 비교할 때 유용하게 쓰인다.

시리즈속성 설명
var aSeries = createSeries('any sereis','bar',data1); 시리즈타입을 'bar'로 지정
aSeries.s_color='rgba(242,159,6,1)'; 그라이언트의 첫번째 색값을 가져와서 똑같이 맞춰주면 범례에 색값이 표시됨
차트속성 설명
var colours = [
{red: 242,green: 159,blue: 6},
{red: 254,green: 240,blue: 216}
];
option.colours = colours;
var colourGradientObject =
new ColourGradient(0,option.series[0]['s_data'].length, colours); 		colours배열에 0번째 컬러부터 1번째 컬러까지 option.series[0]['s_data'].length등분하여 그라디언트를 적용함
option.colorObj = colourGradientObject; 그라이언트 OBJECT지정
option.gradientColorUse = true; 그라이언트 컬러를 사용함

4. Random Color(Single column)

시리즈속성 설명
var aSeries = createSeries('any sereis','bar',data1); 시리즈타입을 'bar'로 지정
aSeries.s_color='rgba(242,159,6,1)'; randomColorUse에서는 시리즈컬러를 지정해도 범례에 색값이 표시되지않음
차트속성 설명
option.randomColorUse = true; 랜덤으로 색을 지정함
option.crossLineShow_hori = false; 툴팁라인 수평선은 사용안함

11. Stacked column

시리즈속성 설명
var maxSum;
var min;
var aSeries1 = createSeries('any sereis A','bar',chartUtil.formatData_profile2(data11_1,false,null,null,null));
...
maxSum = aSeries1.s_max;
min = aSeries1.s_min;
var aSeries2 = createSeries('any sereis B','bar',chartUtil.formatData_profile2(data11_2,false,null,null,null));
...
maxSum += aSeries2.s_max;
min = aSeries2.s_min;
var aSeries3 = createSeries('any sereis C','bar',chartUtil.formatData_profile2(data11_3,false,null,null,null));
...
maxSum += aSeries3.s_max;
min = aSeries3.s_min;
aSeries1.s_max = maxSum;
aSeries2.s_max = maxSum;
aSeries3.s_max = maxSum;
aSeries1.s_min = min;
aSeries2.s_min = min;
aSeries3.s_min = min; 		각시리즈의 max값을 모두 더해서 최종max값으로 재셋팅한다.
차트속성 설명
option.barVertical = true; bar시리즈를 수직으로 연결하여 세운다.

7. Multi column

시리즈속성 설명
var aSeries = createSeries('any sereis C','bar',data7_2); bar시리즈 지정
var aSeries2 = createSeries('any sereis D','line',data7_3); line시리즈지정
aSeries2.s_axis = 'R'; 'R'은 우측Y축,'L'은 좌측Y축(디폴트)을 사용함
aSeries2.s_shapeColor = aSeries2.s_color; 시리즈라인의 컬러와 포인트도형의 컬러를 일치시킨다.
차트속성 설명
option.bothYAxis = true; Y축을 좌측,우측 2개를 사용합니다.
option.toFixedY = 2; 축의 숫자표시를 소수2째자리까지 표시함
